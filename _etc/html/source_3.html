// Sprite textures func addSpritesToNode(node: SKSpriteNode) { var textures = [SKTexture]() for i in 1...10 { let name = NSString(format: "slice%.2d", i) as String let texture = SKTexture(imageNamed: name) textures.append(texture) } let sprite = SKAction.animateWithTextures(textures, timePerFrame: 0.05) node.runAction(SKAction.repeatActionForever(sprite)) }
